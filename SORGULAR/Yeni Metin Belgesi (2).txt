SELECT
CRD.CODE[CARİ KODU],
CRD.DEFINITION_[CARİ ÜNVANI],
VNO.TAXNR[VERGİ NO],
VRD.TAXOFFICE[VERGİ DAİRESİ],
ADR.ADDR1[ADRES 1],
TEL.TELNRS1[TEL NO],
MHK.CODE[Muhasebe(Hesap)kodu]

FROM LG_002_CLCARD I 
LEFT JOIN LG_002_CLCARD  CRD ON CRD.CODE=I.LOGICALREF 
LEFT JOIN LG_002_CLCARD  EFM  ON I.EDINO=EFM.DEFINITION_
LEFT JOIN LG_002_CLCARD  VNO  ON I.TAXNR=VNO.TRADINGGRP AND VNO.TAXNR='1234567890'
LEFT JOIN LG_002_CLCARD  VRD  ON VRD.TRADINGGRP=I.TAXOFFICE AND VRD.TAXOFFICE='ŞİŞLİ'
LEFT JOIN LG_002_CLCARD  ADR  ON I.ADDR1=ADR.TRADINGGRP 
LEFT JOIN LG_002_CLCARD  TEL  ON I.TELNRS1=TEL.TRADINGGRP 
LEFT JOIN LG_002_CLCARD MHK  ON I.EDINO=MHK.CODE

WHERE CRD.CODE='0000000000000001' AND ADR.ADDR1='BARBAROS HAYRETTİN PAŞA MAH' AND TEL.TELNRS1='5315200146'
